#DB 생성
CREATE DATABASE communis_pharmacy;

#User 생성
CREATE USER 'communis'@'localhost' 
IDENTIFIED BY 'pharmacy';

#User 권한부여
GRANT ALL privileges 
ON communis_pharmacy.*
TO 'communis'@'localhost'
WITH GRANT option;
FLUSH PRIVILEGES;

#인기 약품 Table
CREATE TABLE pill_rank (
pill_rank_id BIGINT NOT NULL,
pill_rank_type VARCHAR(256) NOT NULL,
pill_name VARCHAR(256) NOT NULL,
pill_dibs BIGINT DEFAULT 0,
pill_sell BIGINT DEFAULT 0,
PRIMARY KEY(pill_rank_id)
);
